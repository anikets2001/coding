//program to find smallest and largest element in an array

#include<iostream>
using namespace std;

//function to find minimum Element in array
void smallestElement(int arr[],int size)
{
    int min=arr[0];
    for(int i=0;i<size;i++)
    {
      if(arr[i]<min)
      {
          int temp=arr[i];
          arr[i]=min;
          min=temp;
          
      }
    }
    cout<<min<<endl;
}

//function to find largest element in array
void largestElement(int arr[],int size)
{
    int max=arr[0];
    for(int i=0;i<size;i++)
    {
        if(arr[i]>max)
        {
            int temp=arr[i];
            arr[i]=max;
            max=temp;
        }
    }
    cout<<max;
}

//main function 
int main()
{
    int arr[]={2,3,4,7,15,13,5,9};
    int size=sizeof(arr)/sizeof(arr[0]);
    
    cout<<"Smallest element in array: ";
    smallestElement(arr,size);
    
    cout<<"Largest Element in array: ";
    largestElement(arr,size);
    
}
