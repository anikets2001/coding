//program to reverse the original array
#include<iostream>
using namespace std;

//function to reverse the array
void reverse(int arr[],int size)
{
    int j=size-1;
    
    
    int i=0;
   while(i<j)
   {
       int temp=arr[i];
       arr[i]=arr[j];
       arr[j]=temp;
       i++;
       j--;
   }
}

//main function
int main()
{
    int arr[]={1,2,3,4,5,6,7};
    int size=sizeof(arr)/sizeof(arr[0]);
    
    cout<<"original array: ";
    for(int i=0;i<size;i++)
    {
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    
    cout<<"reversed array: ";
    reverse(arr,size);
    for(int i=0;i<size;i++)
    {
        cout<<arr[i]<<" ";
    }
    
}
