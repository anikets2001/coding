//program to find peak element indices in an array
#include<iostream>
using namespace std;

//function to find peak elements
void peak(int arr[],int length)
{
    //if only one element in array
    if(length==0)
    {
        cout<<0<<" ";
    }
    
    //if first element is greater than the second
     if(arr[0]>arr[1])
    {
        cout<<0<<" ";
    }
    
    //check for all other elements except last element
    for(int i=1;i<length-2;i++)
    {
        if(arr[i]>arr[i-1] && arr[i]>arr[i+1])
        {
            cout<<i<<" ";
        }
    }
    
    //check for last element
     if(arr[length-1]>arr[length-2])
    {
        cout<<length-1<<" ";
    }
}
int main()
{
    int arr[]={3,5,2,6,3,7,4,8};
    int length=sizeof(arr)/sizeof(arr[0]);
    
    
    cout<<"indices of peak elements are:"<<endl;
    peak(arr,length);
}
