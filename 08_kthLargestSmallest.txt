//program to find kth largest and smallest element in an array
#include<iostream>
using namespace std;


//function to sort array
void sort(int arr[],int size)
{
    for(int i=0;i<size;i++)
    {
        for(int j=i+1;j<size;j++)
        {
            if(arr[j]<arr[i])
            {
                int temp=arr[i];
                arr[i]=arr[j];
                arr[j]=temp;
            }
        }
    }
}

//function to find kth smallest element 
void smallest(int arr[],int size,int k)
{
    sort(arr,size);
    
    //printing kth smallest element from array
    cout<<arr[k-1];
    
}


//function to find kth largest element
void largest(int arr[],int size,int k)
{
    sort(arr,size);
    
    cout<<arr[size-k];
}


//main function 
int main()
{
    int k;
    cout<<"enter the value of k: ";
    cin>>k;
    
    int arr[]={2,5,4,3,15,18,13,9,8,6};
    int size=sizeof(arr)/sizeof(arr[0]);
    
    cout<<k<<"th smallest element in array is: ";
    smallest(arr,size,k);
    
    cout<<endl;
    
    cout<<k<<"th largest element in array is: ";
    largest(arr,size,k);
}
