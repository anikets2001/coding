//program to find kth largest element 
#include <iostream>
using namespace std;

int main()
{
    int arr[]={2,5,1,7,4,15,3,17};
    int k=4;
    
   
   //calculating array size 
    int length=sizeof(arr)/sizeof(arr[0]);
    
     
    for(int i=0;i<length-1;i++)
    {
        for(int j=0;j<length-1-i;j++)
        {
            if(arr[j]>arr[j+1])
            {
           int temp=arr[j];
            arr[j]=arr[j+1];
            arr[j+1]=temp;
        }
      
    }
   
  }
  cout<<arr[length-k];
}
