//kaden's algorithm 
#include <iostream>
using namespace std;

//function to find maximum sum subarray
void kaden(int arr[],int length)
{
    int curr_sum=0;
    int max_sum=0;
    
    for(int i=0;i<length;i++)
    {
        curr_sum+=arr[i];
        
        if(curr_sum>max_sum)
        {
            max_sum=curr_sum;
        }
        
        if(curr_sum<0)
        {
            curr_sum=0;
        }
    }
    cout<<max_sum;
}
int main()
{
    int arr[]={-5,4,6,-3,4,-1};
    int length=sizeof(arr)/sizeof(arr[0]);
  
    //function call
    kaden(arr,length);
    
   
}
