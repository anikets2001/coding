#include<iostream>
using namespace std;


void reversearr(int arr[],int beg,int end)
{
   
    while(beg<end)
    {
       int temp = arr[beg];
        arr[beg] = arr[end];
        arr[end] = temp;
        beg++;
        end--;
    }
}
void printarr(int arr[],int size)
{
    //printing array
    for(int i=0;i<size;i++)
    {
      cout<<arr[i]<<" ";   
    }
    cout<<endl;
   
}
int main()
{
    //declaring array
    int arr[]={1,2,3,4,5};
    
    //calculating array size
    int length=sizeof(arr)/sizeof(arr[0]);
    
    //printing original array
    cout<<"printing priginal array"<<endl;
    printarr(arr,length);
   
   
    reversearr(arr,0,length-1);
    
    //printing reversed array
    cout<<"printing reversed array"<<endl;
    printarr(arr,length);
    
    
    
}
