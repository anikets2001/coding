//program to find peak element in array
#include<iostream>
using namespace std;


void peak(int arr[],int size)
{
    if(size==1)
    {
        cout<<0<<" ";
    }
    if(arr[0]>arr[1])
    {
        cout<< 0 <<" ";
    }
    if(arr[size-1]>arr[size-2])
    {
      cout<<size-1<<" ";
    }
    
    for(int i=1;i<size-1;i++)
    {
        if(arr[i]>arr[i-1] && arr[i]>arr[i+1])
        {
           cout<<i<<" ";
        }
    }
    
}


//main function
int main()
{
    int arr[]={3,5,2,6,3,7,4,8};
    int size=sizeof(arr)/sizeof(arr[0]);
    
    cout<<"indices with peak elements are: ";
    peak(arr,size);
}
