//program to find missing number in array
#include<iostream>
using namespace std;

//function to find missing number
void missing(int arr[],int length)
{
    int total=(length+1)*(length+2)/2;
    
    for(int i=0;i<length;i++)
    {
        total-=arr[i];
    }
    
    cout<<"missing number is "<<total;
}
int main()
{
    int arr[]={4,1,2,6,3};
    int length=sizeof(arr)/sizeof(arr[0]);
    
    //function call
    missing(arr,length);
}
