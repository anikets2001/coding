//program to reverse the array
#include<iostream>
using namespace std;

void reverse(int arr[],int length)
{
    
    int beg=0;
    int end=length-1;
    
    while(beg<end)
    {
        int temp=arr[beg];
        arr[beg]=arr[end];
        arr[end]=temp;
        beg++;
        end--;
        
    }
}
int main()
{
    int arr[]={1,2,3,4,5};
    int length=sizeof(arr)/sizeof(arr[0]);
    
    
    
    cout<<"original array"<<endl;
    for(int i=0;i<length;i++)
    {
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    
    //function call
    reverse(arr,length);
    
    cout<<"reversed array"<<endl;
    for(int i=0;i<length;i++)
    {
        cout<<arr[i]<<" ";
    }
}
