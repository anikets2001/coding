//DNF sort 
#include <iostream>
using namespace std;

//function to sap values
void swap(int arr[],int i,int j)
{
    int temp=arr[i];
    arr[i]=arr[j];
    arr[j]=temp;
}
//function to perform dnf sort
void dnfSort(int arr[],int size)
{
    int mid=0,low=0,high=size-1;
    while(mid<=high)
    {
        if(arr[mid]==0)
        {
            swap(arr,mid,low);
            mid++; low++;
        }
        else if(arr[mid]==1)
        {
            mid++;
        }
        else if(arr[mid]==2)
        {
            swap(arr,mid,high);
            high--;
        }
    }
}
int main()
{
    int arr[]={1,2,0,0,1,1,2,1,2,1,0,0};
    int size=sizeof(arr)/sizeof(arr[0]);
    
    dnfSort(arr,size);
    
    for(int i=0;i<size;i++)
    {
        cout<<arr[i]<<" ";
    }
    
    
    
}
